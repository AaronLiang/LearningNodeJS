# local debug
$ node debug [debug.js]

# remote debug
$ node --debug[=port] [debug.js]
$ node --debug-brk[=port] [debug.js]

# in Eclipse
update link:  http://chromedevtools.googlecode.com/svn/update/dev/

# node-inspector
$ npm install -g node-inspcetor
$ node --debug-brk=5858 debug.js
$ node-inspector
# open Chrome, Chromium or Safari, go to http://127.0.0.1:8080/debug?port=5858
# node-inspector can not be used in Firefox or Internet Explorer, since it uses WebKit Web Inspector
