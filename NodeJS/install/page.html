# in Ubuntu:
$ sudo apt-get install node.js
$ sudo apt-get install npm
$ npm install nvm or npm install n # nvm or n is the version control for node.js

# in Windows
download windows installer from nodejs.org, and install. npm is included.

# install supervisor # for debug
$ sudo npm install -g supervisor # in Ubuntu
# run supervisor
$ superviosr app.js
